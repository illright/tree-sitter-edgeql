WITH MODULE test
GROUP Issue
USING B :=  Issue.status.name
BY B
INTO Issue
UNION _ := (
    sum := sum(<int64>Issue.number),
    status := B,
)
FILTER
    _.sum > 5
ORDER BY
    B;
